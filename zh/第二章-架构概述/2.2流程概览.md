# Nanobot 消息流转流程详解

## 一、概述

本文通过一个完整的 **WhatsApp 消息处理流程**，深入剖析 Nanobot 内部的消息流转机制。从用户发送消息到收到 AI 回复，整个流程涉及多个模块的协作。

### 阶段一：消息接收（Channels 层）

#### 1.1 WhatsApp 渠道初始化

```python
# channels/whatsapp.py 简化版
class WhatsAppChannel(BaseChannel):
    """WhatsApp Business API 渠道实现"""
    
    name = "whatsapp"
    
    def __init__(self, config: WhatsAppConfig, bus: MessageBus):
        super().__init__(config, bus)
        self.config = config
        self.api = WhatsAppAPI(
            phone_id=config.phone_id,
            token=config.token,
        )
    
    async def start(self) -> None:
        """启动 WhatsApp Webhook 监听"""
        # 设置 Webhook 接收消息
        await self.api.set_webhook(callback=self._on_message)
    
    async def _on_message(self, payload: dict) -> None:
        """收到 WhatsApp 消息回调"""
        # 解析消息内容
        message = payload["entry"][0]["changes"][0]["value"]["messages"][0]
        
        sender_id = message["from"]           # 发送者手机号
        chat_id = message["from"]             # 单聊时 chat_id = sender_id
        content = message["text"]["body"]     # 消息文本
        
        # 创建标准化消息对象
        inbound_msg = InboundMessage(
            channel="whatsapp",
            sender_id=sender_id,
            chat_id=chat_id,
            content=content,
            timestamp=datetime.now(),
        )
        
        # 发送到消息总线
        await self._handle_message(inbound_msg)
```

```python
# channels/base.py
class BaseChannel(ABC):
    
    async def _handle_message(self, msg: InboundMessage) -> None:
        """统一消息处理入口"""
        # 检查白名单
        if self.config.allow_from:
            if msg.sender_id not in self.config.allow_from:
                logger.warning(f"Unauthorized message from {msg.sender_id}")
                return  # 忽略未授权消息
        
        # 发送到消息总线
        await self.bus.publish_inbound(msg)
```

### 阶段二：消息入队（Bus 层）

```python
# bus/queue.py
class MessageBus:
    """异步消息总线"""
    
    def __init__(self):
        # 双向队列：入站消息队列
        self._inbound: asyncio.Queue[InboundMessage] = asyncio.Queue()
        # 双向队列：出站消息队列
        self._outbound: asyncio.Queue[OutboundMessage] = asyncio.Queue()
    
    async def publish_inbound(self, message: InboundMessage) -> None:
        """发布入站消息"""
        await self._inbound.put(message)
        logger.debug(f"Published inbound: {message.channel}:{message.sender_id}")
    
    async def consume_inbound(self) -> InboundMessage:
        """消费入站消息（阻塞等待）"""
        return await self._inbound.get()
    
    async def publish_outbound(self, message: OutboundMessage) -> None:
        """发布出站消息"""
        await self._outbound.put(message)
    
    async def consume_outbound(self) -> OutboundMessage:
        """消费出站消息（阻塞等待）"""
        return await self._outbound.get()
```

### 阶段三：Agent 处理（核心层）

#### 3.1 Agent Loop 主循环

```python
# agent/loop.py
class AgentLoop:
    """Agent 核心处理引擎"""
    
    async def run(self) -> None:
        """运行 Agent 主循环"""
        self._running = True
        await self._connect_mcp()  # 连接 MCP 服务器
        
        while self._running:
            try:
                # 从总线获取消息（超时 1 秒，用于检查停止命令）
                msg = await asyncio.wait_for(
                    self.bus.consume_inbound(), 
                    timeout=1.0
                )
            except asyncio.TimeoutError:
                continue
            
            # 处理 /stop 命令
            if msg.content.strip().lower() == "/stop":
                await self._handle_stop(msg)
            else:
                # 创建异步任务处理消息
                task = asyncio.create_task(self._dispatch(msg))
                self._active_tasks.setdefault(msg.session_key, []).append(task)
    
    async def _dispatch(self, msg: InboundMessage) -> None:
        """分发消息处理"""
        try:
            await self._process_message(msg)
        except Exception as e:
            logger.exception(f"Error processing message: {e}")
            # 发送错误回复
            await self.bus.publish_outbound(OutboundMessage(
                channel=msg.channel,
                chat_id=msg.chat_id,
                content=f"Error: {str(e)}",
            ))
```

#### 3.2 消息处理核心逻辑

```python
# agent/loop.py
async def _process_message(self, msg: InboundMessage) -> None:
    """处理单条消息"""
    
    # ========== 步骤 1: 获取或创建会话 ==========
    session = self.sessions.get_or_create(msg.session_key)
    
    # ========== 步骤 2: 构建上下文 ==========
    # System Prompt + 历史消息 + 记忆 + 技能
    messages = self.context.build_messages(
        session=session,
        user_content=msg.content,
    )
    
    # ========== 步骤 3: 运行 ReAct 循环 ==========
    response, tools_used = await self._run_agent_loop(
        initial_messages=messages,
    )
    
    # ========== 步骤 4: 保存到会话历史 ==========
    session.add_message({"role": "user", "content": msg.content})
    session.add_message({"role": "assistant", "content": response})
    self.sessions.save(session)
    
    # ========== 步骤 5: 发送响应 ==========
    await self.bus.publish_outbound(OutboundMessage(
        channel=msg.channel,
        chat_id=msg.chat_id,
        content=response,
        reply_to=msg.metadata.get("message_id"),
    ))
```

#### 3.3 ReAct 循环详解

```python
# agent/loop.py
async def _run_agent_loop(
    self,
    initial_messages: list[dict],
) -> tuple[str | None, list[str]]:
    """
    运行 ReAct（Reasoning + Acting）循环
    
    核心逻辑：
    1. 调用 LLM 获取响应
    2. 如果响应包含工具调用 → 执行工具 → 继续循环
    3. 如果响应是最终回复 → 返回结果
    """
    messages = initial_messages
    iteration = 0
    final_content = None
    tools_used: list[str] = []
    
    while iteration < self.max_iterations:  # 最大 40 轮迭代
        iteration += 1
        
        # 获取工具定义（用于 Function Calling）
        tool_defs = self.tools.get_definitions()
        
        # ========== 调用 LLM ==========
        response = await self.provider.chat_with_retry(
            messages=messages,
            tools=tool_defs,
            model=self.model,
        )
        
        # ========== 情况 1: 需要调用工具 ==========
        if response.has_tool_calls:
            # 记录思考过程
            if response.content:
                logger.info(f"Thought: {response.content[:200]}")
            
            # 将助手消息（含工具调用）添加到上下文
            messages = self.context.add_assistant_message(
                messages, 
                response.content, 
                [tc.to_openai_tool_call() for tc in response.tool_calls],
            )
            
            # 执行每个工具调用
            for tool_call in response.tool_calls:
                tools_used.append(tool_call.name)
                
                # 执行工具
                result = await self.tools.execute(
                    tool_call.name, 
                    tool_call.arguments,
                )
                
                # 将工具结果添加到上下文
                messages = self.context.add_tool_result(
                    messages,
                    tool_call.id,
                    tool_call.name,
                    result,
                )
            
            # 继续循环，让 LLM 基于工具结果继续推理
            continue
        
        # ========== 情况 2: 最终回复 ==========
        else:
            final_content = response.content
            break
    
    return final_content, tools_used
```

### 阶段四：上下文构建

```python
# agent/context.py
class ContextBuilder:
    """构建 LLM 上下文（System Prompt + Messages）"""
    
    # 启动时加载的引导文件
    BOOTSTRAP_FILES = [
        "AGENTS.md",    # Agent 角色定义
        "SOUL.md",      # 人格/语气定义
        "USER.md",      # 用户信息
        "TOOLS.md",     # 工具说明
        "IDENTITY.md",  # 身份标识
    ]
    
    def build_messages(
        self,
        session: Session,
        user_content: str,
    ) -> list[dict]:
        """构建完整的消息列表"""
        
        messages = []
        
        # ========== 1. System Prompt ==========
        system_prompt = self._build_system_prompt()
        messages.append({"role": "system", "content": system_prompt})
        
        # ========== 2. 长期记忆 ==========
        memory = self.memory.load()
        if memory:
            messages.append({
                "role": "system", 
                "content": f"[Memory]\n{memory}",
            })
        
        # ========== 3. 会话历史 ==========
        for msg in session.get_recent_messages(limit=20):
            messages.append({
                "role": msg["role"],
                "content": msg["content"],
            })
        
        # ========== 4. 当前用户消息 ==========
        messages.append({"role": "user", "content": user_content})
        
        return messages
    
    def _build_system_prompt(self) -> str:
        """构建 System Prompt"""
        parts = []
        
        # 加载引导文件
        for filename in self.BOOTSTRAP_FILES:
            filepath = self.workspace / filename
            if filepath.exists():
                parts.append(filepath.read_text())
        
        # 加载技能说明
        skills = self.skills.load_descriptions()
        if skills:
            parts.append(f"[Skills]\n{skills}")
        
        return "\n\n".join(parts)
```

以 `web_search` 工具为例：

```python
# agent/tools/web.py
class WebSearchTool(BaseTool):
    """网络搜索工具"""
    
    name = "web_search"
    description = "Search the web for information"
    
    parameters = {
        "type": "object",
        "properties": {
            "query": {
                "type": "string",
                "description": "Search query",
            },
        },
        "required": ["query"],
    }
    
    def __init__(self, api_key: str | None = None, proxy: str | None = None):
        self.api_key = api_key
        self.proxy = proxy
    
    async def execute(self, query: str) -> str:
        """执行搜索"""
        # 调用 Brave Search API
        async with httpx.AsyncClient(proxy=self.proxy) as client:
            response = await client.get(
                "https://api.search.brave.com/res/v1/web/search",
                headers={"X-Subscription-Token": self.api_key},
                params={"q": query, "count": 5},
            )
            data = response.json()
        
        # 格式化结果
        results = []
        for item in data.get("web", {}).get("results", []):
            results.append(f"- {item['title']}: {item['url']}\n  {item['description']}")
        
        return "\n".join(results) if results else "No results found."
```

### 阶段七：Channel Manager 分发

```python
# channels/manager.py
class ChannelManager:
    """渠道管理器"""
    
    def __init__(self, config: Config, bus: MessageBus):
        self.config = config
        self.bus = bus
        self.channels: dict[str, BaseChannel] = {}
        
        # 初始化所有启用的渠道
        self._init_channels()
    
    def _init_channels(self) -> None:
        """初始化渠道实例"""
        # 自动发现所有渠道类
        for name, channel_class in discover_channels().items():
            channel_config = getattr(self.config.channels, name, None)
            if channel_config and channel_config.enabled:
                self.channels[name] = channel_class(channel_config, self.bus)
                logger.info(f"Initialized channel: {name}")
    
    async def start(self) -> None:
        """启动所有渠道"""
        # 启动每个渠道的监听
        for channel in self.channels.values():
            asyncio.create_task(channel.start())
        
        # 启动出站消息分发器
        self._dispatch_task = asyncio.create_task(self._dispatch_outbound())
    
    async def _dispatch_outbound(self) -> None:
        """分发出站消息到对应渠道"""
        while True:
            msg = await self.bus.consume_outbound()
            
            # 根据 channel 名称找到对应渠道
            if channel := self.channels.get(msg.channel):
                try:
                    await channel.send(msg)
                except Exception as e:
                    logger.exception(f"Failed to send message: {e}")
            else:
                logger.warning(f"Unknown channel: {msg.channel}")
```

## 五、关键设计要点

### 5.1 异步解耦

```python
# 消息总线实现异步解耦
# Channels 不需要等待 Agent 处理完成
# Agent 也不需要关心消息来自哪个渠道

# Channel 侧：发送后立即返回
await self.bus.publish_inbound(msg)  # 非阻塞

# Agent 侧：独立消费处理
msg = await self.bus.consume_inbound()  # 阻塞等待
```

## 六、流程总结

| 阶段 | 模块 | 核心操作 | 关键代码 |
|------|------|----------|----------|
| 1 | Channels | 接收消息、格式转换、权限检查 | `whatsapp.py:_on_message()` |
| 2 | Bus | 消息入队 | `queue.py:publish_inbound()` |
| 3 | Agent | 获取会话、构建上下文 | `loop.py:_process_message()` |
| 4 | Agent | ReAct 循环 | `loop.py:_run_agent_loop()` |
| 5 | Tools | 工具执行 | `registry.py:execute()` |
| 6 | Bus | 消息出队 | `queue.py:publish_outbound()` |
| 7 | Channels | 发送回复 | `whatsapp.py:send()` |
